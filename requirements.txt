# VTK RAG - All Dependencies

-r retrieval-pipeline/requirements.txt

-r grounding-prompting/requirements.txt

# Stage 5: LLM Generation (anthropic, openai, google-generativeai)
-r llm-generation/requirements.txt

# Stage 6: Post-Processing (no dependencies)
-r post-processing/requirements.txt

# API Validation (MCP for hallucination detection)
-r api-mcp/requirements.txt

# Evaluation (VTK for code execution)
-r evaluation/requirements.txt

# Visual Testing (Docker, Pillow, scikit-image, opencv)
-r visual_testing/requirements.txt
